\documentclass{minimal}
\newcounter{beer}
\newcommand{\verses}[1]{
    \setcounter{beer}{#1}
    \par\noindent
    \arabic{beer} bottles of beer on the wall,\\
    \arabic{beer} bottles of beer!\\
    Take one down, pass it around---\\
    \addtocounter{beer}{-1}
    \arabic{beer} bottles of beer on the wall!\\
    \ifnum#1>0
        \verses{\value{beer}}
    \fi
}
\begin{document}
\verses{99}
\end{document}
